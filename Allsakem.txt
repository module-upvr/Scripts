local Saken = {}

local Player = game:GetService("Players").LocalPlayer
local Run = game:GetService("RunService")
local PlayerG = Player.PlayerGui
local Killers = game:GetService("ReplicatedStorage").Assets.Killers:GetChildren() 
local Survivors = game:GetService("ReplicatedStorage").Assets.Survivors:GetChildren()
local Sprinting = require(game:GetService("ReplicatedStorage").Systems.Character.Game.Sprinting)
local ragdolls = require(game:GetService("ReplicatedStorage").Modules.Ragdolls) 
local blind = require(game:GetService("ReplicatedStorage").Modules.StatusEffects.Blindness)
local Undetect = require(game:GetService("ReplicatedStorage").Modules.StatusEffects.Undetectable)
local ChatConfig = game:GetService("TextChatService").ChatWindowConfiguration
local Landing = game:GetService("ReplicatedStorage").Systems.Character.Miscellaneous:FindFirstChild("SlowOnLanding")
local Able = require(game:GetService("ReplicatedStorage").Assets.Survivors.TwoTime.Behavior).Abilities
local Util = require(game:GetService("ReplicatedStorage").Modules.Util)
local PlayerFolder = game.workspace.Players

local IsCr = false
local crouchThread = nil

function Saken.Esp()
    local function createESP(object, color)
        if object:FindFirstChild("SakenLight") then return end
        local highlight = Instance.new("Highlight")
        highlight.FillTransparency = 0.9
        highlight.OutlineTransparency = 0.8
        highlight.FillColor = color
        highlight.OutlineColor = color
        highlight.Parent = object
        highlight.Name = "SakenLight"
        return highlight
    end

    local plrESP
    plrESP = function(object, color)
        if not object or object:FindFirstChild("SakenLight") then return end
        if object:GetAttribute("Username") == Player.Name then return end

        local highlight = Instance.new("Highlight")
        highlight.Name = "SakenLight"
        highlight.FillTransparency = 0.9
        highlight.OutlineTransparency = 0.8
        highlight.FillColor = color
        highlight.Parent = object  
    end

    local function updateESP()
        for _, v in ipairs(workspace.Map.Ingame:GetDescendants()) do
            if v.Name == "Medkit" and v:IsA("Tool") then
                createESP(v, Color3.new(1, 1, 0))
            elseif v.Name == "BloxyCola" and v:IsA("Tool") then
                createESP(v, Color3.new(0, 0, 1))
            elseif v.Name == "Generator" and v:IsA("Model") and v.Parent.Name ~= "Instances" then
                createESP(v, Color3.new(1, 0, 1))
            end
        end

        for _, v in ipairs(PlayerFolder.Survivors:GetChildren()) do
            plrESP(v, Color3.new(0, 1, 0))
        end

        for _, v in ipairs(PlayerFolder.Killers:GetChildren()) do
            plrESP(v, Color3.new(1, 0, 0))
        end
    end
    
    updateESP()
    Player.CharacterAdded:Connect(updateESP)
    Run.Heartbeat:Connect(function()
        for _, v in ipairs(PlayerFolder.Survivors:GetChildren()) do
            plrESP(v, Color3.new(0, 1, 0))
        end

        for _, v in ipairs(PlayerFolder.Killers:GetChildren()) do
            plrESP(v, Color3.new(1, 0, 0))
        end
    end)
end

function Saken.AutoGen() 
    local function solvgen()
        for _, v in pairs(workspace.Map.Ingame.Map:GetDescendants()) do
            if v:IsA("RemoteEvent") then
                v:FireServer() 
            end
        end
    end

    if getgenv().loop then
        getgenv().loop:Disconnect() 
        getgenv().loop = nil 
    end

    local nextActionTime = 0
    local delayTime = function() 
        return Random.new():NextNumber(2.5, 3) 
    end

    getgenv().loop = Run.Heartbeat:Connect(function()
        local ui = PlayerG:FindFirstChild("PuzzleUI")
        if ui then
            if nextActionTime == 0 then
                nextActionTime = os.clock() + delayTime() 
            end
            if os.clock() >= nextActionTime then
                solvgen() 
                nextActionTime = os.clock() + delayTime() 
            end
        else
            if nextActionTime ~= 0 then
                nextActionTime = 0
            end
        end
    end)
end

function Saken.Config() 
    Sprinting.DefaultConfig.MinStamina = -5400

    for _, Kill in ipairs(Killers) do 
        if not Kill:FindFirstChild("Config") then continue end
        require(Kill.Config).Speed =  13
        require(Kill.Config).SprintSpeed = 32
        if Kill.Name == "Noli" then
            require(Kill.Config).VoidRushDashSpeed = 70
            require(Kill.Config).VoidRushTurnSpeed = math.huge
            require(Kill.Config).VoidRushInitialTurnDuration = 0
            require(Kill.Config).VoidRushInitialTurnMult = 0
        elseif Kill.Name == "Slasher" then
            require(Kill.Config).EnragedSpeed = 30
        elseif Kill.Name == "1x1x1x1" then
            require(Kill.Config).Speed =  13
            require(Kill.Config).SprintSpeed = 30.75
        end
    end 

    for _, Surv in ipairs(Survivors) do 
        if not Surv:FindFirstChild("Config") then continue end
        require(Surv.Config).Speed = 16
        require(Surv.Config).SprintSpeed = 30
        if Surv.Name == "Veeronica" then
            require(Surv.Config).Sk8TurnControl = 6
        elseif Surv.Name == "TwoTime" then
            require(Surv.Config).CrouchSpeed = 15
            require(Surv.Config).CrouchRunSpeed = 30
        end
    end
    
    local function stopTimer()
        if crouchThread then
            task.cancel(crouchThread)
            crouchThread = nil
        end
    end

    local function DoLunge(arg1)
        local Lunge = Util:ApplyVelocity(arg1.Rig, 310)
        game.TweenService:Create(Lunge, TweenInfo.new(0.2), {
    	 					["LineVelocity"] = Lunge.LineVelocity / 3
      					}):Play()
    			  		game.Debris:AddItem(Lunge, 0.2) 
    end

    Able.Dagger.Callback = function(arg1)
        local Anims = arg1.Animations
        if IsCr then
            Anims.LungeStart:Play(0)
            task.delay(Anims.LungeStart.Length, function()
                Anims.LungeLoop:Play(0)
            end) 
            task.delay(0.6, function()
                Anims.LungeEnd:Play() 
                Anims.LungeStart:Stop(0)
                Anims.LungeLoop:Stop(0) 
            end) 
            IsCr = false
            stopTimer()
        else
            Anims.Stab:Play(0)
            task.delay(0.15, function() 
                DoLunge(arg1) 
            end) 
        end
    end

    Able.Crouch.Callback = function(arg1)
        stopTimer() 
        IsCr = true
        crouchThread = task.delay(15, function()
            if IsCr then
                IsCr = false
                crouchThread = nil
            end
        end)
    end
end

function Saken.Function() 
    local function AnimSaken(Rig) 
        local PC = Rig
        if not PC then return end

        if PC:WaitForChild("SpeedMultipliers"):FindFirstChild("DirectionalMovement") then
            PC.SpeedMultipliers.DirectionalMovement:Destroy()
        end
        
        local Invis = Instance.new("Animation", PC) 
        Invis.AnimationId = "rbxassetid://75804462760596"
        Invis.Name = "Invisible"

        local Jerk = Instance.new("Animation", PC)
        Jerk.AnimationId = "rbxassetid://72042024"
        Jerk.Name = "Goon"

        local Block = Instance.new("Animation", PC) 
        Block.AnimationId = "rbxassetid://72722244508749"
        Block.Name = "FakeBlock"

        local Animator = PC:WaitForChild("Humanoid"):WaitForChild("Animator")
        local JerkAnim = Animator:LoadAnimation(Jerk) 
        JerkAnim.Looped = true
        JerkAnim.Priority = Enum.AnimationPriority.Action4
        local BlockAnim = Animator:LoadAnimation(Block) 
        BlockAnim.Priority = Enum.AnimationPriority.Action4

        local Gui = Instance.new("ScreenGui", PlayerG) 
        Gui.Name = "SakenGui"
        Gui.ResetOnSpawn = true

        local TextButton = Instance.new("TextButton", Gui) 
        TextButton.Size = UDim2.new(0.1, 0, 0.1, 0) 
        TextButton.Text = "Goon"

        local TextButton2 = Instance.new("TextButton", Gui) 
        TextButton2.Size = UDim2.new(0.1, 0, 0.1, 0) 
        TextButton2.Position = UDim2.new(0.1, 0, 0, 0) 
        TextButton2.Text = "FakeBlock"

        local TextButton3 = Instance.new("TextButton", Gui) 
        TextButton3.Size = UDim2.new(0.1, 0, 0.1, 0) 
        TextButton3.Position = UDim2.new(0.2, 0, 0, 0) 
        TextButton3.Text = "Ragdoll"

        local Bool, Bool2 = false, false

        TextButton3.MouseButton1Click:Connect(function() 
            if not Bool2 then
                PC.Humanoid.PlatformStand = true
                require(game:GetService("ReplicatedStorage").Modules.Sounds):Play("ragdolledLol", {Parent = PC.PrimaryPart}) 
                Bool2 = true
                pcall(function() ragdolls.EnableRagdoll(PC) end) 
            else
                ragdolls.DisableRagdoll(PC)
                PC.Humanoid.PlatformStand = false
                Bool2 = false
            end
        end) 

        TextButton2.MouseButton1Click:Connect(function() BlockAnim:Play() end) 

        TextButton.MouseButton1Click:Connect(function() 
            if not Bool then JerkAnim:Play() Bool = true else JerkAnim:Stop() Bool = false end
        end) 
    end

    AnimSaken(Player.Character) 
    Player.CharacterAdded:Connect(function(Rig) 
        AnimSaken(Rig)
    end) 

    blind.Applied = function() end
    blind.Removed = function() end

    Undetect.Applied = function(arg1) 
        if arg1.Character then
            local InvisAnim = arg1.Character.Humanoid.Animator:LoadAnimation(arg1.Character.Invisible) 
            InvisAnim:Play(0) 
            InvisAnim:AdjustSpeed(0) 
        end
    end

    Undetect.Removed = function(arg1)
        if arg1.Character then
            local Humanoid = arg1.Character:FindFirstChildOfClass("Humanoid")
            local Animator = Humanoid and Humanoid:FindFirstChildOfClass("Animator")
            
            if Animator then
                local targetId = arg1.Character.Invisible.AnimationId
                for _, track in pairs(Animator:GetPlayingAnimationTracks()) do
                    if track.Animation and string.find(track.Animation.AnimationId, targetId) then
                        track:Stop()
                    end
                end
            end
        end
    end

    if Landing then pcall(function() Landing:Destroy() end) end

    ChatConfig.Enabled = true

    ChatConfig:GetPropertyChangedSignal("Enabled"):Connect(function()
        if not ChatConfig.Enabled then
            ChatConfig.Enabled = true
        end
    end)
end

return Saken
